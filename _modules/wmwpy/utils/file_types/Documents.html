[docs]    def read(this, mime: str, extension: str, rawdata: io.BytesIO, **kwargs):
        tmp = tempfile.mkstemp(suffix = '.db')
        os.write(tmp[0], rawdata.getvalue())
        os.close(tmp[0])
        
        source = sqlite3.connect(tmp[1])
        dest = sqlite3.connect(':memory:')
        source.backup(dest)
        
        source.close()
        
        if os.path.exists(tmp[1]):
            os.remove(tmp[1])
        
        return dest

    

[docs]    def save(this, data: sqlite3.Connection) -> bytes:
        if isinstance(data, sqlite3.Connection):
            tmp = tempfile.mkstemp(suffix = '.db')
            os.close(tmp[0])
            
            dest = sqlite3.connect(tmp[1])
            data.backup(dest)
            
            dest.close()
            
            with open(tmp[1], 'rb') as file:
                rawdata = file.read()
            
            if os.path.exists(tmp[1]):
                os.remove(tmp[1])
            
            return rawdata
        
        return super().save(data)
